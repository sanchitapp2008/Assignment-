# Demonstration of different Python Data Types

# 1. Integer
a = 10
print("Integer value:", a)
print("Type:", type(a))

# 2. Float
b = 10.5
print("\nFloat value:", b)
print("Type:", type(b))

# 3. String
c = "Hello Python"
print("\nString value:", c)
print("Type:", type(c))

# 4. Boolean
d = True
print("\nBoolean value:", d)
print("Type:", type(d))

# 5. List
e = [1, 2, 3, "Apple", "Banana"]
print("\nList value:", e)
print("Type:", type(e))

# 6. Tuple
f = (10, 20, 30, "Orange")
print("\nTuple value:", f)
print("Type:", type(f))

# 7. Dictionary
g = {"Name": "Sanchita", "Age": 18, "Course": "Engineering"}
print("\nDictionary value:", g)
print("Type:", type(g))

# 8. Set
h = {1, 2, 3, 4, 5}
print("\nSet value:", h)
print("Type:", type(h))
